iOS - Swiftйқўеҗ‘еҚҸи®®зј–зЁӢпјҲдәҢпјү

ж¬ўиҝҺе…іжіЁеҫ®дҝЎе…¬дј—еҸ·пјҡ[FSAе…Ёж ҲиЎҢеҠЁ рҹ‘Ӣ]

дёҠдёҖзҜҮж–Үз« iOS - Swift йқўеҗ‘еҚҸи®®зј–зЁӢпјҲдёҖпјүе·Із»ҸеҜ№Swiftзҡ„йқўеҗ‘еҚҸи®®зј–зЁӢеҒҡдәҶд»Ӣз»ҚпјҢжҺҘдёӢжқҘиҜҘзҜҮж–Үз« е°ҶдҪҝз”Ёйқўеҗ‘еҚҸи®®ејҖеҸ‘(POP)жқҘеҒҡдёӢе®һйҷ…зҡ„еә”з”Ё

еңЁе®һйҷ…ејҖеҸ‘дёӯпјҢиҮӘе®ҡд№үViewеҹәжң¬дёҠжҳҜеҝ…йЎ»зҡ„пјҢзӣёдҝЎиҝҷеҜ№жҲ‘д»¬жқҘиҜҙйғҪжҳҜжҜ”иҫғз®ҖеҚ•пјҢдёҚиҝҮжҲ‘д»¬иҝҳжҳҜжқҘеӣһйЎҫдёҖдёӢдёӢ~

# йқўеҗ‘еҜ№иұЎејҖеҸ‘

1 ж–°е»әдёҖдёӘUIViewзҡ„FirstTypeView

2 еҲӣе»әдёҖдёӘViewзҡ„xibж–Үд»¶

3 и®ҫзҪ®xibеҜ№еә”зҡ„classиҝӣиЎҢз»‘е®ҡ

4 еңЁFirstTypeView.swift дёӯе®һзҺ°дёҖдёӘзұ»ж–№жі•пјҢж–№дҫҝжҲ‘д»¬еӨ–йғЁз”ЁxibжқҘеҲқе§ӢеҢ–FirstTypeView

import UIKit

class FirstTypeView: UIView {
    
}

extension FirstTypeView {
    class func loadFromNib() -> FirstTypeView {
        return Bundle.main.loadNibNamed("\(self)", owner: nil, options: nil)?.first as! FirstTypeView
    }
}

еңЁеӨ–йғЁеҸӘиҰҒи°ғз”ЁFirstTypeViewзҡ„loadFromNibж–№жі•е°ұеҸҜд»ҘеҲқе§ӢеҢ–дёҖдёӘViewжқҘдҪҝз”ЁдәҶгҖӮеҘҪпјҢзҺ°еңЁеҸҲжңүдёҖдёӘзұ»SecondTypeViewпјҢд№ҹжҳҜиҰҒжұӮдҪҝз”ЁxibжқҘеҲқе§ӢеҢ–viewгҖӮиҝҷж—¶жҲ‘д»¬е°ұдјҡжғіпјҢдёҖж ·зҡ„еҠ иҪҪxibзҡ„ж–№жі•пјҢйӮЈжҲ‘д»¬е°ұжҠҠе®ғжҠҪеҸ–еҮәжқҘж”ҫеҲ°зҲ¶зұ»е°ұеҸҜд»ҘдәҶгҖӮиҝҷйҮҢзҡ„зҲ¶зұ»д»ҘBaseView.swiftдёәдҫӢ зҲ¶зұ»зҡ„дё»иҰҒе®һзҺ°д»Јз Ғ

extension BaseView {
    class func loadFromNib() -> BaseView {
        return Bundle.main.loadNibNamed("\(self)", owner: nil, options: nil)?.first as! BaseView
    }
}

йӮЈжҲ‘д»¬зҡ„FirstTypeViewе’ҢSecondTypeViewеҸӘйңҖиҰҒзӣҙжҺҘз»§жүҝдәҺBaseViewе°ұеҸҜд»ҘдәҶпјҢеңЁе…¶е®ғең°ж–№еҲқе§ӢеҢ–viewд№ҹеҫҲж–№дҫҝ

let firstView = FirstTypeView.loadFromNib()
view.addSubview(firstView)

let secondView = SecondTypeView.loadFromNib()
view.addSubview(secondView)

еҘҪпјҢзҺ°еңЁFirstTypeViewйҮҢеЈ°жҳҺдәҶдёҖдёӘеұһжҖ§nameпјҢSecondTypeViewеЈ°жҳҺзҡ„еұһжҖ§дёәageпјҢеҒҮеҰӮжҲ‘д»¬зҺ°еңЁиҰҒдҪҝз”Ёеҗ„иҮӘеҜ№еә”зҡ„еұһжҖ§пјҢиҝҷж—¶жҳҜзӣҙжҺҘзӮ№дёҚеҮәжқҘзҡ„пјҢйңҖиҰҒе…ҲиҝӣиЎҢејәиҪ¬

let firstView = FirstTypeView.loadFromNib() as! FirstTypeView
firstView.name = "LXF"
view.addSubview(firstView)

let secondView = SecondTypeView.loadFromNib() as! SecondTypeView
secondView.age = 100
view.addSubview(secondView)

дҪҶжҳҜиҝҷж ·и§үеҫ—дёҚжҳҜеҫҲж–№дҫҝпјҢиҝҳйңҖиҰҒиҝӣиЎҢејәиҪ¬пјҢжҲ‘д»¬иғҪдёҚиғҪеңЁBaseViewйҮҢйқўжҗһе®ҡе®ғе‘ўпјҹеҰӮжһңжҳҜSwift 2.x зҡ„иҜқжҳҜеҸҜд»Ҙзҡ„

extension BaseView {
    class func loadFromNib() -> Self { // жіЁж„ҸиҝҷйҮҢжҳҜеӨ§еҶҷзҡ„S
        return Bundle.main.loadNibNamed("\(self)", owner: nil, options: nil)?.first as! Self
    }
}

дҪҶжҳҜзҺ°еңЁSwift 3.0е·Із»ҸдёҚж”ҜжҢҒиҝҷз§ҚеҶҷжі•дәҶпјҢдјҡжҠҘй”ҷгҖӮиҝҷдёӘж—¶еҖҷеҰӮжһңдҪҝз”Ёйқўеҗ‘еҚҸи®®зҡ„ејҖеҸ‘е°ұеҫҲж–№дҫҝдәҶгҖӮ

# йқўеҗ‘еҚҸи®®ејҖеҸ‘

е°ҶBaseViewеҲ йҷӨпјҢFirstTypeViewе’ҢSecondTypeViewж”№еӣһз»§жүҝдәҺUIView

1 ж–°е»әдёҖдёӘSwiftж–Үд»¶ Nibloadable.swift

2 е®һзҺ°еҚҸи®®ж–№жі•

еҚҸи®®дёӯдёҚе…Ғи®ёе®ҡд№үзұ»ж–№жі•пјҢйңҖж”№дёәйқҷжҖҒж–№жі•

import UIKit

protocol Nibloadable {
    
}

extension Nibloadable {
    static func loadFromNib() -> Self {
        return Bundle.main.loadNibNamed("\(self)", owner: nil, options: nil)?.first as! Self
    }
}

3 йҒөе®ҲеҚҸи®®

class SecondTypeView: UIView, Nibloadable {
    var age: Int = 10
}

иҝҷж ·е°ұеҸҜд»ҘдәҶпјҢиҖҢдё”дҪ еңЁи°ғз”ЁloadFromNibж–№жі•ж—¶еҸҜд»ҘеҸ‘зҺ°пјҢзұ»еһӢжҳҜеҜ№еә”дёҠзҡ„

еҘҪдәҶпјҢйқўеҗ‘еҚҸи®®ејҖеҸ‘зҡ„еә”з”Ёе°ұи®°еҪ•еҲ°иҝҷйҮҢпјҢеёҢжңӣиғҪеё®еҠ©еҲ°еӨ§е®¶пјҒ
