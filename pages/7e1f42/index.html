		super();
		this.list = list;
	}

	@Override
	public Object next() {
		Object obj = null;
		if (hasNext()) {
			obj = list.get(index);
			index++;
		}
		return obj;
	}

	@Override
	public boolean hasNext() {
		return index < list.size();
	}
